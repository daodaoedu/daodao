# Admin 前端頁面設計規格

## 概述

在 daodao-f2e monorepo 中新增 `apps/admin` 應用，提供統一的後台管理介面。

## 技術選型

```yaml
框架: Next.js 15 (App Router)
UI 庫: @daodao/ui (shadcn/ui 為基礎)
狀態管理: SWR (swr-openapi)
圖表: Recharts
表格: @tanstack/react-table
表單: react-hook-form + zod
通知: sonner
```

---

## 頁面結構

```
apps/admin/src/app/
├── layout.tsx                     # 根佈局 (側邊欄 + Header)
├── page.tsx                       # Dashboard 首頁
├── loading.tsx                    # 全局載入狀態
├── error.tsx                      # 全局錯誤邊界
│
├── services/                      # 服務監控
│   ├── page.tsx                  # 服務總覽
│   ├── daodao-server/
│   │   ├── page.tsx             # Express 服務詳情
│   │   ├── health/page.tsx      # 健康檢查歷史
│   │   └── logs/page.tsx        # 日誌查看
│   └── ai-backend/
│       ├── page.tsx             # FastAPI 服務詳情
│       ├── llm/page.tsx         # LLM 狀態與用量
│       └── tasks/page.tsx       # Celery 任務監控
│
├── database/                      # 資料庫管理
│   ├── page.tsx                  # 資料庫總覽
│   ├── postgresql/
│   │   ├── page.tsx             # 表列表
│   │   ├── [table]/page.tsx     # 表詳情
│   │   └── query/page.tsx       # SQL 查詢工具
│   ├── redis/
│   │   ├── page.tsx             # Redis 總覽 (兩個實例)
│   │   ├── server/
│   │   │   └── page.tsx        # Server Redis 詳情
│   │   └── ai-backend/
│   │       └── page.tsx        # AI Redis 詳情
│   ├── mongodb/
│   │   ├── page.tsx             # Collection 列表
│   │   └── [collection]/page.tsx
│   └── qdrant/
│       ├── page.tsx             # Collection 列表
│       └── [collection]/page.tsx
│
├── api-docs/                      # API 文檔
│   ├── page.tsx                  # 文檔入口
│   ├── server/page.tsx          # daodao-server Swagger
│   └── ai-backend/page.tsx      # ai-backend Swagger
│
└── users/                         # 用戶管理
    ├── page.tsx                  # 用戶列表
    └── [id]/page.tsx            # 用戶詳情
```

## 路由保護

```typescript
// middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  const token = request.cookies.get('token');

  if (!token) {
    return NextResponse.redirect(new URL('/login', request.url));
  }

  // 驗證 admin 權限
  // ...

  return NextResponse.next();
}

export const config = {
  matcher: ['/((?!login|_next|favicon.ico).*)'],
};
```
